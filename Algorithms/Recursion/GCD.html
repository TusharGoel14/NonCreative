GCD(Greatest Common Divisor)

            
                import java.util.*;
                class main{   
                    public static void main(String[] args){
                       int ans=gcd(4,6);
                       System.out.println(ans);
                    }
                    
                    static int gcd(int divident, int divisor){
                        // Base Case
                        if(divisor==0)
                        return divident;
                        // Recursive Case
                        else
                        return gcd(divisor,divident%divisor);
                    }
                
                }
