Fast Exponentiation using Recursion

            
                import java.util.*;
                class main{   
                    public static void main(String[] args){

                        int a=3,b=5,mod=7;
                        
                        // need to calcualte 3^5%7
                        System.out.println(pow(a,b,mod));
                    }

                    static int pow(int a,int b,int mod) {
                        //Base Case
                        if(b==0){
                            return 1;
                        }
                        else if(b==1){
                            return a%mod;
                        }
                        // Recursive Case
                        //when the b is even
                        else if(b%2==0){
                            int y=pow(a,b/2,mod)%mod;
                            return (y*y)%mod;
                        }
                        //when b is odd
                        else {
                            int y=pow(a,b-1,mod)%mod;
                            return ((a%mod)*y)%mod;
                        }
                    }

                }
