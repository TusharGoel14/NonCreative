                import java.util.*;
                class main{   
                    public static void main(String[] args){
                       //palindrome with the help of recursion 
                       int n=23432;
                       int p=rev(n,0);
                       System.out.println(p);
                       if(n==p)
                       System.out.print("palindrome");
                       else
                       System.out.println("not palindrome");
                    }
                    static int rev(int n, int p){
                        //Base case
                        if(n==0)
                        return p;
                        //recursive case
                        p=rev(n/10,p*10+n%10);
                        //n=234 === 23 === 2==0
                        //p=0*10+4==4*10+3=43*10+2=432
                        return p;
                    }             
                }
