CLASS (Basic Class Implementation)

            
                import java.util.*;
                class main{   
                    public static void main(String[] args){
                        pair p=new pair(2,5);    
                        System.out.println(p.first+" "+p.second); 
                    }

                    static class pair{
                        int first;
                        int second;
                        pair(int first, int second){
                            this.first=first;
                            this.second=second;
                        }
                    }
                }
